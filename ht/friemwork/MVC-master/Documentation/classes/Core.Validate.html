    Validates user input.

    This class can be run in "single" mode or "mass" mode. In both situations it is designed to be human-readable, and provide a simple API to basic validation.

    Single mode example

    if (Core\Validate::is('cjhill@gmail.com', 'email')) {
    // Houston, we have a valid email address
}

    Mass mode example

    $form = new Validate(array(
    'Email address' => array(
        'value' => 'cjhill@gmail.com',
        'tests' => array(
            'required' => true,
            'is'       => 'email'
        )
    ),
    'Age' => array(
        'value' => 20,
        'tests' => array(
            'between' => array('min' => 13, 'max' => 99)
        )
    ),
    'Password' => array(
        'value' => 's3cr37',
        'tests' => array(
            'required'      => true,
            'required_with' => array('Email address')
            'length'        => array('min' => 8)
        )
    )
));

if (! $form->isValid()) {
    // Errors
    var_dump($validate->getErrors());
}
